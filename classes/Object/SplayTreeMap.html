static VALUE splaytree_max_key(VALUE self) {
        splaytree *tree = get_tree_from_self(self);
        splaytree_node *node;
        
        if(!tree->root)
                return Qnil;
        
        node = tree->root;
        while (node->right)
                node = node->right;
        
        return node->key;
}

static VALUE splaytree_min_key(VALUE self) {
        splaytree *tree = get_tree_from_self(self);
        splaytree_node *node;
        
        if(!tree->root)
                return Qnil;
        
        node = tree->root;
        while (node->left)
                node = node->left;
        
        return node->key;
}
